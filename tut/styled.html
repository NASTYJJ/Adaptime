Note that you can drag appointments between the two plancanvasses, and that the green plancanvas has a smaller 'snap' than the brown one. If you drop appointments outside the canvasses, nothing happens since you have to assign a dom element (possibly the document.body) as a 'garbage' element.
The code: 
function ignore(event)
{
    // just completely ignore events
}

function main()
{
	var d=document.getElementById("s1");
    var cl_b = new CanvasLayout(3600000,"brownstyle");
    var cl_g = new CanvasLayout(60000,"greenstyle");
    
    var dbgtest = document.getElementById("debug");

    if (dbgtest) {
        global.debug = new Debug(dbgtest);
    }

    var t1 = new AppType(-1,"doing nothing","#117788", null, 1.0,true);
    var types = [ t1 ];

    start = last_midnight();
    end = start + 24 * 3600000;
	d.style.width="800px";
	d.style.height="400px";
    var pc1 = new PlanCanvas(d,start,end,cl_b,ignore,types,null);
	pc1.resize(800,200);
    var pc2 = new PlanCanvas(d,start,end,cl_g,ignore,types,null);
	pc2.xywh(0,0,800,200);
}
