The grey rectangle is one PlanCanvas. It shows 24 hours of any day. The standard timeformatting function that displays times in the Appointment works with microseconds, so that's what i used for the start and endtime (and the snap value). Try creating appointments with the right mouse button. also move them around you may notice that overlapping is forbidden by default.
The basic movements are creation, resizing and movement. Code : 
// reusable variables
var ddc = new DragDropContext();

function ignore(e)
{
    // just completely ignore events
}

function main()
{
	var d=document.getElementById("s1");
    var cl = new CanvasLayout(30*60000,null);

    var t1 = new AppType(1,"doing nothing","orange","apptype_0",1.0,true);
    var types = [ t1 ];

    start = 0;
    end = start + 24*60*60000;
    d.style.background="grey";
    d.style.width="800px";
    d.style.height="100px";
    var pc = new PlanCanvas(d,start,end,cl,ignore,types,ddc,0);
}
